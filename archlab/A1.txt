    call sum_list

    xorq %rax, %rax
    testq %rdi, %rdi
    je .L1
.L2:    
    mrmovq (%rdi), %r8
    addq %r8, %rax
    mrmovq 8(%rdi), %rdi
    testq %rdi, %rdi
    jne .L2
.L1:
    ret
